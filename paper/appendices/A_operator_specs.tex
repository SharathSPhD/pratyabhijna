\section{Operator Specifications}\label{app:operator_specs}

This appendix gives the full equations and signatures for each operator. The reference implementation is in \texttt{src/pce/operators/} and is exercised by the test suite in \texttt{tests/operators/}.

\subsection{$\textsc{Cit}$}
\[
\textsc{Cit}(\text{prompt}, \tau, \text{seed}) \to \text{Candidate}
\]
The single-sample LLM generation. Sampler defaults: $\tau{=}0.9$, top-$p{=}0.95$, top-$k{=}50$. Determinism: torch RNG on CPU (MPS does not support per-device generators) feeds \texttt{multinomial}; the device runs the model itself.

\subsection{$\textsc{\=Ananda}$}
\[
\textsc{\=Ananda}(c, \kappa, R) = w_{\text{coh}}\cos(\phi(c), \phi(\kappa.\text{text})) + w_{\text{div}}\bigl(1 - \max_{r\in R}\cos(\phi(c), \phi(r))\bigr) + w_{\text{form}}\,F(c)
\]
where $\kappa$ is the constraint, $R$ is a retrieval set, and $F(c)$ is a form-fidelity heuristic (e.g., line-count proxy for haiku). Default weights: $(w_{\text{coh}}, w_{\text{div}}, w_{\text{form}}) = (0.5, 0.3, 0.2)$.

\subsection{$\textsc{Icch\=a}$}
$K$ candidate generations under a constraint-augmented prompt. The constraint is inlined as natural prose to avoid bracket-leakage in small-LM substrates (we calibrated this in Phase~6: the original \texttt{[Constraint: ...]} bracketed form leaked the literal bracket back into surfaces).

\subsection{$\textsc{Apohana}$}
\[
a_i = 1 - \max_{j} \cos\bigl(\phi(c_i), \phi(m_j)\bigr), \quad m_j \in \text{must-avoid}
\]
The contrastive exclusion score; bounded in $[0, 2]$ but in practice $[0,1]$ because cosine of normalised embeddings is in $[-1, 1]$. The implementation clips to $[0, 1]$ for numerical safety.

\subsection{$\textsc{J\~n\=ana}$ (Bayesian Model Reduction)}
Let pseudo-counts $\alpha_i = \alpha_0 + \beta a_i$ for $i\in 1\dots K$, $\alpha_0 = 1$, $\beta = 4$. The Dirichlet--multinomial evidence under the rich model (uniform prior) is
\[
\log p(o\mid m_F) = \log\Gamma(K) - \sum_i \log\Gamma(1) + \sum_i (\alpha_i - 1)\log a_i.
\]
The reduced model concentrates the prior on $i^* = \arg\max_i \alpha_i$ via $\alpha_i' = \alpha_i + 1$ at $i^*$ only. The free-energy delta is
\[
\Delta F = \log p(o\mid m_R) - \log p(o\mid m_F),
\]
computed using \texttt{scipy.special.gammaln} for numerical stability. We do not import \texttt{pymdp}'s JAX backend because we need exact CPU determinism for the audit log.

\subsection{$\textsc{Kriy\=a}$}
Three modes:
\begin{itemize}
\item \texttt{verbatim} returns the chosen candidate's surface unchanged.
\item \texttt{polish} normalises whitespace and trims trailing assistant-style babble (\texttt{Human:}, \texttt{Assistant:}, etc.).
\item \texttt{claude\_polish} (opt-in) sends the surface to Claude Code with a tight ``preserve content, polish surface'' prompt; not used in the headline benchmarks.
\end{itemize}

\subsection{$\textsc{Vimar\'sa}$}
\[
\text{event} = \mathbb{1}\bigl[\text{aspect\_multiplicity} \lor \text{low\_novelty} \lor \text{negative\_residual\_}\Delta F\bigr]
\]
\textsc{aspect\_multiplicity} is true iff at least two aspect-target embeddings have cosine $\geq 0.40$ to the surface embedding (calibrated in Phase~6, see ADR-002 for the calibration trace).
\textsc{low\_novelty} is true iff the maximum cosine to any retrieval-set element is $\geq 0.85$.
\textsc{negative\_residual\_}$\Delta F$ is true iff the BMR $\Delta F$ recorded on the chosen candidate is $\geq 0$ (i.e., the prior was not yet falsified --- the cascade should re-enter).

\paragraph{The novelty signal.} \textsc{Vimar\'sa.novelty} is computed as $1 - \max_{r\in R}\cos(\phi(\text{surface}), \phi(r))$, clipped to $[0, 1]$. It is recorded in the audit log alongside \texttt{event}.
